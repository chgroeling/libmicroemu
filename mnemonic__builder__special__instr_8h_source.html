75 static void BuildMsrMrs(TContext &mctx, const u8 &mask, const u8 &SYSm) {
76 const auto SYSm_7_3 = Bm32::ExtractBits1R<7U, 3U>(SYSm);
77 switch (SYSm_7_3) {
78 case 0b00000U: {
79 if (mask & 0x1) {
80 mctx.builder.AddString("APSR_g"); // Application Program Status Register
81 } else {
82 mctx.builder.AddString("APSR_nzcvq"); // Application Program Status Register
83 }
84 break;
85 }
86 case 0b00001U: {
87 const auto SYSm_2_0 = Bm32::ExtractBits1R<2U, 0U>(SYSm);
88 switch (SYSm_2_0) {
89 case 0b000U:
90 mctx.builder.AddString("MSP"); // Main Stack Pointer
91 break;
92 case 0b001U:
93 mctx.builder.AddString("PSP"); // Process Stack Pointer
94 break;
95 default:
96 mctx.builder.AddString("undefined");
97 break;
98 }
99 break;
100 }
101 case 0b00010U: {
102 const auto SYSm_2_0 = Bm32::ExtractBits1R<2U, 0U>(SYSm);
103 switch (SYSm_2_0) {
104 case 0b000U:
105 mctx.builder.AddString("PRIMASK"); // Priority Mask
106 break;
107 case 0b001U:
108 mctx.builder.AddString("BASEPRI"); // Base Priority
109 break;
110 case 0b010U:
111 mctx.builder.AddString("BASEPRI_MAX"); // Base Priority Max
112 break;
113 case 0b011U:
114 mctx.builder.AddString("FAULTMASK"); // Fault Mask
115 break;
116 case 0b100U:
117 mctx.builder.AddString("CONTROL"); // Control
118 break;
119 default:
120 mctx.builder.AddString("undefined");
121 break;
122 }
123 } break;
124 default:
125 mctx.builder.AddString("undefined");
126 break;
127 }
