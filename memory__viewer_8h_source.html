47

58 void print_section(TProcessorStates &pstates, const me_adr_t vadr_begin, const me_adr_t vadr_end,
59 const u32 alignment, u32 indent, bool skip_read) {
60
61 for (me_adr_t ivadr = vadr_begin; ivadr < vadr_end; ivadr++) {
62 // Print address first
63 if (ivadr % alignment == 0u) {
64 for (u32 spaces = 0u; spaces < indent; ++spaces) {
65 printf(" ");
66 }
67 printf("%08x|", ivadr);
68 }
69
70 // Then the byte
71 if (!skip_read) {
72 auto res = mem_.template Read<u8>(pstates, ivadr);
73 if (res.IsErr()) {
74 printf("xx");
75 } else {
76 printf("%02x", res.content);
77 }
78 } else {
79 printf(" ");
80 }
81
82 // Following by an delimiter
83 printf(" ");
84
85 // After x bytes add a newline
86 // ... special condition do not add newline at the end
87 if ((ivadr % alignment == alignment - 1) && (ivadr != vadr_end - 1)) {
88 printf("\n");
89 }
90 }
