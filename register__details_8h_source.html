7template <unsigned last_bit, unsigned first_bit> static constexpr u32 BitMask() {
8 if constexpr (last_bit == sizeof(u32) * 8U - 1U) {
9 return ~0U << first_bit;
10 } else {
11 return ((1U << (last_bit - first_bit + 1U)) - 1U) << first_bit;
12 }
13}
