23 template <typename TArg>
24 static Result<ExecResult> Call(TInstrContext &ictx, const InstrFlagsSet &iflags,
25 const TArg &arg_n, const u32 &imm32, const TArg &arg_t) {
26 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
27 const bool is_index = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kIndex)) != 0U;
28 const bool is_add = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kAdd)) != 0U;
29
30 ExecFlagsSet eflags{0x0U};
31 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
32
33 if (!condition_passed) {
34 It::ITAdvance(ictx.pstates);
35 Pc::AdvanceInstr(ictx.pstates, is_32bit);
36 return Ok(ExecResult{eflags});
37 }
38
39 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
40 const me_adr_t offset_addr = is_add == true ? rn + imm32 : rn - imm32;
41 const me_adr_t address = is_index == true ? offset_addr : rn;
42
43 TRY_ASSIGN(data, ExecResult, TOp::Read(ictx, address));
44
45 const bool is_wback = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kWBack)) != 0U;
46 if (is_wback == true) {
47 Reg::WriteRegister(ictx.pstates, arg_n.Get(), offset_addr);
48 }
49
50 if (arg_t.Get() == RegisterId::kPc) {
51 // When the given address was unaligend the behaviour is
52 // unpredtictable
53 if ((address & 0x3U) == 0U) {
54 It::ITAdvance(ictx.pstates);
55 TRY(ExecResult, Pc::LoadWritePC(ictx.pstates, ictx.bus, data));
56
57 return Ok(ExecResult{eflags});
58 } else {
59 return Err<ExecResult>(StatusCode::kScExecutorUnpredictable);
60 }
61 } else {
62 Reg::WriteRegister(ictx.pstates, arg_t.Get(), data);
63 }
64 It::ITAdvance(ictx.pstates);
65 Pc::AdvanceInstr(ictx.pstates, is_32bit);
66
67 return Ok(ExecResult{eflags});
68 }
