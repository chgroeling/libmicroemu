77 static inline void ITAdvance(TProcessorStates &pstates) {
78 auto istate = SReg::template ReadRegister<SpecialRegisterId::kIstate>(pstates);
79 const auto istate_2_0 = istate & IstateRegister::kItBit2to0Msk;
80
81 if (istate_2_0 == 0x0U) {
82 istate = 0x0U;
83 } else {
84 const auto next_istate_4_0 =
85 ((istate & IstateRegister::kItBit4to0Msk) << 1U) & IstateRegister::kItBit4to0Msk;
86 istate = (istate & ~IstateRegister::kItBit4to0Msk) | next_istate_4_0;
87 }
88
89 SReg::template WriteRegister<SpecialRegisterId::kIstate>(pstates, istate);

92 static inline Result<u32> CurrentCond(const TProcessorStates &pstates) {
93 // • For the T1 and T3 encodings of the Branch instruction shown in B on
94 // page A7-205, it returns the 4-bit cond field of the encoding.
95 // --> This is handled by the encoder by Using ConiditionPassed(cond)
96
97 // • For all other Thumb instructions:
98 auto istate = SReg::template ReadRegister<SpecialRegisterId::kIstate>(pstates);
99
100 const auto it_3_0 = istate & IstateRegister::kItBit3to0Msk;
101
102 if (it_3_0 != 0x0U) {
103 const auto it_7_4 = (istate & IstateRegister::kItBit7to4Msk) >> IstateRegister::kItBit4Pos;
104 return Ok<u32>(it_7_4);
105 }
106
107 if (istate == 0x0U) {
108 return Ok<u32>(0xEU); // b1110
109 }
110
111 // Otherwise, execution of the instruction is UNPREDICTABLE.
112 return Err<u32>(StatusCode::kScExecutorUnpredictable);
113 }

169 static inline bool ConditionPassed(const TProcessorStates &pstates, u8 cond) {
170 // see Armv7-M Architecture Reference Manual Issue E.e p.178 - 179
171 bool result{false};
172
173 auto cond_3_1 = (cond & IstateRegister::kItBit3to1Msk) >> IstateRegister::kItBit1Pos;
174 switch (cond_3_1) {
175 case 0b000U: { // EQ - Equal or NE - Not equal
176 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
177 result = (apsr & ApsrRegister::kZMsk) == ApsrRegister::kZMsk;
178 break;
179 }
180 case 0b001U: { // CS - Carry set or CC - Carry clear
181 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
182 result = (apsr & ApsrRegister::kCMsk) == ApsrRegister::kCMsk;
183 break;
184 }
185 case 0b010U: { // MI - Minus, negative or PL - Plus, positive or zero
186 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
187 result = (apsr & ApsrRegister::kNMsk) == ApsrRegister::kNMsk;
188 break;
189 }
190 case 0b011U: { // VS - Overflow or VC - No overflow
191 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
192 result = (apsr & ApsrRegister::kVMsk) == ApsrRegister::kVMsk;
193 break;
194 }
195 case 0b100U: { // HI - Unsigned higher or LS - Unsigned lower or same
196 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
197 result = ((apsr & ApsrRegister::kCMsk) == ApsrRegister::kCMsk) &&
198 ((apsr & ApsrRegister::kZMsk) != ApsrRegister::kZMsk);
199 break;
200 }
201 case 0b101U: { // GE - Signed greater than or equal or LT - Signed less than
202 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
203 result = ((apsr & ApsrRegister::kNMsk) >> ApsrRegister::kNPos) ==
204 ((apsr & ApsrRegister::kVMsk) >> ApsrRegister::kVPos);
205 break;
206 }
207 case 0b110U: { // GT - Signed greater than or LE - Signed less than or
208 // equal
209 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
210 result = (((apsr & ApsrRegister::kNMsk) >> ApsrRegister::kNPos) ==
211 ((apsr & ApsrRegister::kVMsk) >> ApsrRegister::kVPos)) &&
212 ((apsr & ApsrRegister::kZMsk) != ApsrRegister::kZMsk);
213 break;
214 }
215 case 0b111U: { // AL - Always
216 result = true;
217 break;
218 }
219 default: {
220 // should not happen
221 assert(false);
222 }
223 }
224
225 // Condition flag values in the set '111x' indicate the instruction is
226 // always executed Otherwise, invert condition if necessary.
227
228 // if cond<0> == '1' && cond != '1111' then result = !result;
229 if (((cond & 0x1U) == 0x1U) && (cond != 0xFU)) {
230 result = !result;
231 }
232
233 return result;
234 }

236 static inline Result<bool> ConditionPassed(const TProcessorStates &pstates) {
237 // see Armv7-M Architecture Reference Manual Issue E.e p.179
238
239 const auto r_cond = CurrentCond(pstates);
240 if (r_cond.IsErr()) {
241 return Err<u32, bool>(r_cond);
242 }
243
244 const auto &cond = r_cond.content;
245
246 return Ok(ConditionPassed(pstates, cond));
