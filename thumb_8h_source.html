24 static Result<ThumbImmediateResult> ThumbExpandImm_C(const uint32_t &in, const bool carry_in) {
25 // see Armv7-M Architecture Reference Manual Issue E.e p.139 - 140
26 u32 imm32{0U};
27 bool carry_out{false};
28
29 if (Bm32::Slice1R<11U, 10U>(in) == 0x0U) {
30 switch (Bm32::Slice1R<9U, 8U>(in)) {
31
32 case 0x0U: {
33 imm32 = Bm32::ZeroExtend<u32>(in & 0xFFU);
34 break;
35 }
36 case 0x1U: {
37 if ((in & 0xFFU) == 0x0U) {
38 return Err<ThumbImmediateResult>(StatusCode::kScExecutorUnpredictable);
39 }
40 imm32 = ((in & 0xFFU) << 16U) | (in & 0xFFU);
41 break;
42 }
43 case 0x2U: {
44 if ((in & 0xFFU) == 0x0U) {
45 return Err<ThumbImmediateResult>(StatusCode::kScExecutorUnpredictable);
46 }
47 imm32 = ((in & 0xFFU) << 24U) | ((in & 0xFFU) << 8U);
48 break;
49 }
50 case 0x3U: {
51 if ((in & 0xFFU) == 0x0U) {
52 return Err<ThumbImmediateResult>(StatusCode::kScExecutorUnpredictable);
53 }
54 imm32 = ((in & 0xFFU) << 24U) | ((in & 0xFFU) << 16U) | ((in & 0xFFU) << 8U) |
55 ((in & 0xFFU) << 0U);
56 break;
57 }
58 default:
59 // should not happen
60 assert(false);
61 break;
62 }
63 carry_out = carry_in;
64 } else {
65 const u32 unrotated_value = Bm32::ZeroExtend<u32>(0x80U | (in & 0x7FU));
66 const auto r_ror = Alu32::ROR_C(unrotated_value, Bm32::Slice1R<11U, 7U>(in));
67
68 imm32 = r_ror.result;
69 carry_out = r_ror.carry_out;
70 }
71 return Ok(ThumbImmediateResult{imm32, carry_out});
72 }
