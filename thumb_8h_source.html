23 static Result<ThumbImmediateResult> ThumbExpandImm_C(const uint32_t &in, const bool carry_in) {
24 // see Armv7-M Architecture Reference Manual Issue E.e p.139 - 140
25 u32 imm32{0U};
26 bool carry_out{false};
27
28 if (Bm32::ExtractBits1R<11U, 10U>(in) == 0x0U) {
29 switch (Bm32::ExtractBits1R<9U, 8U>(in)) {
30
31 case 0x0U: {
32 imm32 = Bm32::ZeroExtend<u32>(in & 0xFFU);
33 break;
34 }
35 case 0x1U: {
36 if ((in & 0xFFU) == 0x0U) {
37 return Err<ThumbImmediateResult>(StatusCode::kExecutorUnpredictable);
38 }
39 imm32 = ((in & 0xFFU) << 16U) | (in & 0xFFU);
40 break;
41 }
42 case 0x2U: {
43 if ((in & 0xFFU) == 0x0U) {
44 return Err<ThumbImmediateResult>(StatusCode::kExecutorUnpredictable);
45 }
46 imm32 = ((in & 0xFFU) << 24U) | ((in & 0xFFU) << 8U);
47 break;
48 }
49 case 0x3U: {
50 if ((in & 0xFFU) == 0x0U) {
51 return Err<ThumbImmediateResult>(StatusCode::kExecutorUnpredictable);
52 }
53 imm32 = ((in & 0xFFU) << 24U) | ((in & 0xFFU) << 16U) | ((in & 0xFFU) << 8U) |
54 ((in & 0xFFU) << 0U);
55 break;
56 }
57 default:
58 // should not happen
59 assert(false);
60 break;
61 }
62 carry_out = carry_in;
63 } else {
64 const u32 unrotated_value = Bm32::ZeroExtend<u32>(0x80U | (in & 0x7FU));
65 const auto r_ror = Alu32::ROR_C(unrotated_value, Bm32::ExtractBits1R<11U, 7U>(in));
66
67 imm32 = r_ror.result;
68 carry_out = r_ror.carry_out;
69 }
70 return Ok(ThumbImmediateResult{imm32, carry_out});
71 }
