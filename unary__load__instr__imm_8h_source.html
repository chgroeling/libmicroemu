23 static Result<ExecResult> Call(TInstrContext &ictx, const InstrFlagsSet &iflags, const u32 &imm32,
24 const TArg &arg_t) {
25 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
26
27 ExecFlagsSet eflags{0x0U};
28 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
29
30 if (!condition_passed) {
31 It::ITAdvance(ictx.pstates);
32 Pc::AdvanceInstr(ictx.pstates, is_32bit);
33 return Ok(ExecResult{eflags});
34 }
35
36 const bool is_add = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kAdd)) != 0U;
37
38 // unary loads always refer to the pc
39 const me_adr_t pc = static_cast<me_adr_t>(Reg::ReadPC(ictx.pstates));
40
41 const me_adr_t base = Bm32::AlignDown<4>(pc);
42 const me_adr_t address = is_add == true ? base + imm32 : base - imm32;
43
44 TRY_ASSIGN(data, ExecResult, TOp::Read(ictx, address));
45
46 if (arg_t.Get() == RegisterId::kPc) {
47 // When the given address was unaligend the behaviour is
48 // unpredtictable
49 if ((address & 0x3U) == 0U) {
50 It::ITAdvance(ictx.pstates);
51 TRY(ExecResult, Pc::LoadWritePC(ictx.pstates, ictx.bus, data));
52
53 return Ok(ExecResult{eflags});
54 } else {
55 return Err<ExecResult>(StatusCode::kScExecutorUnpredictable);
56 }
57 } else {
58 Reg::WriteRegister(ictx.pstates, arg_t.Get(), data);
59 }
60 It::ITAdvance(ictx.pstates);
61 Pc::AdvanceInstr(ictx.pstates, is_32bit);
62
63 return Ok(ExecResult{eflags});
64 }
