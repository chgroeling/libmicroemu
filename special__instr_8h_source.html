47 TDelegates &delegates) {
48 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
49
50 ExecFlagsSet eflags{0x0U};
51
52 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
53
54 if (!condition_passed) {
55 It::ITAdvance(ictx.pstates);
56 Pc::AdvanceInstr(ictx.pstates, is_32bit);
57 return Ok(ExecResult{eflags});
58 }

60 TRY_ASSIGN(svc_flags, ExecResult, delegates.Svc(imm32));
61
62 if ((svc_flags & static_cast<SvcFlagsSet>(SvcFlags::kRequestExit)) != 0U) {
63 eflags |= static_cast<ExecFlagsSet>(ExecFlags::kSvcReqExit);
64 } else if ((svc_flags & static_cast<SvcFlagsSet>(SvcFlags::kRequestErrorExit)) != 0U) {
65 eflags |= static_cast<ExecFlagsSet>(ExecFlags::kSvcReqErrorExit);
66 }
67
68 if ((svc_flags & static_cast<SvcFlagsSet>(SvcFlags::kOmitException)) == 0U) {
69 ExcTrig::SetPending(ictx.pstates, ExceptionType::kSVCall);
70 }
71 } else {
72 ExcTrig::SetPending(ictx.pstates, ExceptionType::kSVCall);
73 }
74
75 It::ITAdvance(ictx.pstates);
76 Pc::AdvanceInstr(ictx.pstates, is_32bit);
77 return Ok(ExecResult{eflags});

140 const TArg0 &arg_m, TArg1 arg_n) {
141 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
142
143 ExecFlagsSet eflags{0x0U};
144 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
145
146 if (!condition_passed) {
147 It::ITAdvance(ictx.pstates);
148 Pc::AdvanceInstr(ictx.pstates, is_32bit);

150 }
151
152 const bool is_tbh = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kTbh)) != 0U;
153

159 TRY_ASSIGN(out, ExecResult,

162 halfwords = out;
163 } else {
164 me_adr_t adr = rn + rm;
165 TRY_ASSIGN(out, ExecResult,
166 ictx.bus.template ReadOrRaise<u8>(ictx.pstates, adr,
167 BusExceptionType::kRaisePreciseDataBusError));
168 halfwords = out;
169 }
170
171 const me_adr_t pc = static_cast<me_adr_t>(Reg::ReadPC(ictx.pstates));
172 Pc::BranchWritePC(ictx.pstates, pc + (halfwords << 1U));
173
174 return Ok(ExecResult{eflags});

206 template <typename TArg0, typename TArg1>
207 static Result<ExecResult> Bfi(TInstrContext &ictx, const InstrFlagsSet &iflags,
208 const TArg0 &arg_d, const TArg1 &arg_n, const u8 &lsbit,
209 const u8 &msbit) {
210
211 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
212
213 ExecFlagsSet eflags{0x0U};
214 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
215
216 if (!condition_passed) {
217 It::ITAdvance(ictx.pstates);
218 Pc::AdvanceInstr(ictx.pstates, is_32bit);
219 return Ok(ExecResult{eflags});
220 }
221
222 if (msbit >= lsbit) {
223 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
224 const auto rd = Reg::ReadRegister(ictx.pstates, arg_d.Get());
225
226 const auto src_bitmask = static_cast<u32>(static_cast<u32>(1U) << (msbit - lsbit + 1U)) - 1U;
227 const auto dest_bitmask = static_cast<u32>(
228 ((static_cast<u32>(1U) << (msbit - lsbit + 1U)) - static_cast<u32>(1U)) << lsbit);
229
230 const auto rn_slice = (rn & src_bitmask) << lsbit;
231 const auto rd_result = (rd & ~dest_bitmask) | rn_slice;
232
233 Reg::WriteRegister(ictx.pstates, arg_d.Get(), rd_result);
234 } else {
235 return Err<ExecResult>(StatusCode::kScExecutorUnpredictable);
236 }
237
238 It::ITAdvance(ictx.pstates);
239 Pc::AdvanceInstr(ictx.pstates, is_32bit);
240 return Ok(ExecResult{eflags});
241 }

287 const bool is_add = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kAdd)) != 0U;

389 template <typename TArg0>
390 static Result<ExecResult> Msr(TInstrContext &ictx, const InstrFlagsSet &iflags,
391 const TArg0 &arg_n, const uint8_t mask, const uint8_t SYSm) {
392
393 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
394
395 ExecFlagsSet eflags{0x0U};
396 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
397
398 if (!condition_passed) {
399 It::ITAdvance(ictx.pstates);
400 Pc::AdvanceInstr(ictx.pstates, is_32bit);
401 return Ok(ExecResult{eflags});
402 }
403 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
404 const auto SYSm_7_3 = Bm32::Slice1R<7U, 3U>(SYSm);
405 switch (SYSm_7_3) {
406 case 0b00000U: {
407 if (mask & 0x1) {
408 // APSR_g Application Program Status Register
409 assert(false && "not implemented");
410 } else {
411 // APSR_nzcvq Application Program Status Register
412 assert(false && "not implemented");
413 }
414 break;
415 }
416 case 0b00001U: {
417 const auto SYSm_2_0 = Bm32::Slice1R<2U, 0U>(SYSm);
418 switch (SYSm_2_0) {
419 case 0b000U: {
420 // MSP - Main Stack Pointer
421 SReg::template WriteRegister<SpecialRegisterId::kSpMain>(ictx.pstates, rn);
422 LOG_TRACE(TLogger, "MSR Call - Write main stack pointer: 0x%08X", rn);
423 break;
424 }
425 case 0b001U: {
426 // PSP - Process Stack Pointer
427 SReg::template WriteRegister<SpecialRegisterId::kSpProcess>(ictx.pstates, rn);
428 LOG_TRACE(TLogger, "MSR Call - Write process stack pointer: 0x%08X", rn);
429 break;
430 }
431 default:
432 // undefined
433 assert(false && "undefined");
434 break;
435 }
436 break;
437 }
438 case 0b00010U: {
439 const auto SYSm_2_0 = Bm32::Slice1R<2U, 0U>(SYSm);
440 switch (SYSm_2_0) {
441 case 0b000U:
442 // PRIMASK - Priority Mask
443 assert(false && "not implemented");
444 break;
445 case 0b001U:
446 // BASEPRI - Base Priority
447 assert(false && "not implemented");
448 break;
449 case 0b010U:
450 // BASEPRI_MAX - Base Priority Max
451 assert(false && "not implemented");
452 break;
453 case 0b011U:
454 // FAULTMASK- Fault Mask
455 assert(false && "not implemented");
456 break;
457 case 0b100U: {
458 // CONTROL- Control
459 if (CurrentModeIsPrivileged(ictx.pstates)) {
460 auto control = SReg::template ReadRegister<SpecialRegisterId::kControl>(ictx.pstates);
461 // CONTROL.nPRIV = R[n]<0>;
462 control &= ~ControlRegister::kNPrivMsk;
463 control |= ((rn & 0x1) >> 0U) << ControlRegister::kNPrivPos;
464
465 // TODO: Move to separate file. Function IsThreadMode and IsHandlerMode
466 auto sys_ctrl = SReg::template ReadRegister<SpecialRegisterId::kSysCtrl>(ictx.pstates);
467 const auto exec_mode = sys_ctrl & SysCtrlRegister::kExecModeMsk;
468 const auto is_thread_mode = exec_mode == SysCtrlRegister::kExecModeThread;
469 if (is_thread_mode) {
470 // CONTROL.SPSEL = R[n]<1>;
471 control &= ~ControlRegister::kSpselMsk;
472 control |= ((rn & 0x2U) >> 1U) << ControlRegister::kSpselPos;
473 }
474 LOG_TRACE(TLogger, "MSR Call - Write CONTROL: 0x%08X", control);
475
476 SReg::template WriteRegister<SpecialRegisterId::kControl>(ictx.pstates, control);
477 }
478
479 // if HaveFPExt() then CONTROL.FPCA = R[n]<2>;
480 break;
481 }
482 default:
483 // undefined
484 assert(false && "undefined");
485 break;
486 }
487 } break;
488 default:
489 // undefined
490 assert(false && "undefined");
491 break;
492 }
493
494 It::ITAdvance(ictx.pstates);
495 Pc::AdvanceInstr(ictx.pstates, is_32bit);
496 return Ok(ExecResult{eflags});
497 }

606 template <typename TArg0, typename TArg1, typename TArg2>
607 static Result<ExecResult> Smull(TInstrContext &ictx, const InstrFlagsSet &iflags,
608 const TArg0 &arg_d_lo, const TArg0 &arg_d_hi, const TArg1 &arg_n,
609 const TArg2 &arg_m) {
610
611 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
612
613 ExecFlagsSet eflags{0x0U};
614 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
615
616 if (!condition_passed) {
617 It::ITAdvance(ictx.pstates);
618 Pc::AdvanceInstr(ictx.pstates, is_32bit);
619 return Ok(ExecResult{eflags});
620 }
621
622 const auto rn = static_cast<i32>(Reg::ReadRegister(ictx.pstates, arg_n.Get()));
623 const auto rm = static_cast<i32>(Reg::ReadRegister(ictx.pstates, arg_m.Get()));
624 const u64 result = static_cast<u64>(static_cast<i64>(rn) * static_cast<i64>(rm));
625 const u32 result_lo = static_cast<u32>(result & 0xFFFFFFFFU);
626 const u32 result_hi = static_cast<u32>((result >> 32U) & 0xFFFFFFFFU);
627 Reg::WriteRegister(ictx.pstates, arg_d_lo.Get(), result_lo);
628 Reg::WriteRegister(ictx.pstates, arg_d_hi.Get(), result_hi);
629 It::ITAdvance(ictx.pstates);
630 Pc::AdvanceInstr(ictx.pstates, is_32bit);
631 return Ok(ExecResult{eflags});
632 }

636 template <typename TArg0, typename TArg1, typename TArg2>
637 static Result<ExecResult> Strd(TInstrContext &ictx, const InstrFlagsSet &iflags,
638 const TArg0 &arg_t, const TArg1 &arg_t2, const TArg2 &arg_n,
639 const u32 &imm32) {
640
641 ExecFlagsSet eflags{0x0U};
642 const bool is_wback = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kWBack)) != 0U;
643 const bool is_index = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kIndex)) != 0U;
644 const bool is_add = (iflags & static_cast<InstrFlagsSet>(InstrFlags::kAdd)) != 0U;
645 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
646
647 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
648
649 if (!condition_passed) {
650 It::ITAdvance(ictx.pstates);
651 Pc::AdvanceInstr(ictx.pstates, is_32bit);
652 return Ok(ExecResult{eflags});
653 }
654 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
655
656 const u32 off_adr = is_add == true ? rn + imm32 : rn - imm32;
657 const u32 address = is_index == true ? off_adr : rn;
658
659 const auto rt = Reg::ReadRegister(ictx.pstates, arg_t.Get());
660 TRY(ExecResult, ictx.bus.template WriteOrRaise<u32>(
661 ictx.pstates, address, rt, BusExceptionType::kRaisePreciseDataBusError));
662
663 //---
664 const auto rt2 = Reg::ReadRegister(ictx.pstates, arg_t2.Get());
665 TRY(ExecResult,
666 ictx.bus.template WriteOrRaise<u32>(ictx.pstates, address + 0x4, rt2,
667 BusExceptionType::kRaisePreciseDataBusError));
668
669 if (is_wback == true) {
670 Reg::WriteRegister(ictx.pstates, arg_n.Get(), off_adr);
671 }
672 //---
673
674 It::ITAdvance(ictx.pstates);
675 Pc::AdvanceInstr(ictx.pstates, is_32bit);
676 return Ok(ExecResult{eflags});
677 }
