26 // @brief Copies data from an memory address used by the emulator to an local array.
27 // @param dest_ptr The character array to store the data.
28 // @param dest_len The maximum length of the character array.
29 // @param mem_acs object to access the emulated memory.
30 // @param src_ptr The memory address to read the data from.
31 // @param len The length of data to be read.
32 // @return Result<void> The result of the write operation.
33 template <typename TProcessorStates, typename TBus>
34 static Result<void> CpyFromEmuMem(TProcessorStates &pstates, TBus &bus, char *dest_ptr,
35 std::size_t dest_len, me_adr_t src_ptr, me_size_t src_len) {
36 auto res_len = src_len <= dest_len ? src_len : dest_len;
37
38 for (u32 i = 0U; i < res_len; ++i) {
39 auto r_ch = bus.template Read<u8>(pstates, src_ptr);
40 if (r_ch.IsErr()) {
41 return Err<u8, void>(r_ch);
42 };
43 const auto &ch = r_ch.content;
44 dest_ptr[i] = ch;
45 src_ptr++;
46 }
47
48 return Ok();
49 }

51 // @brief Copies data from an local array to an memory address used by the emulator.
52 template <typename TProcessorStates, typename TBus>
53 static Result<u32> CpyToEmuMem(TProcessorStates &pstates, TBus &bus, me_adr_t dest_ptr,
54 me_size_t dest_len, char *src_ptr, std::size_t src_len) {
55 auto res_len = dest_len <= src_len ? dest_len : src_len;
56
57 for (u32 i = 0U; i < res_len; ++i) {
58 auto r_ch = bus.template Write<u8>(pstates, dest_ptr, src_ptr[i]);
59 if (r_ch.IsErr()) {
60 return Err<void, u32>(r_ch);
61 };
62 dest_ptr++;
63 }
64
65 return Ok<u32>(res_len);
66 }
