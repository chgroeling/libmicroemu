22 template <typename TArg>
23 static Result<ExecResult> Call(TInstrContext &ictx, const InstrFlagsSet &iflags,
24 const TArg &arg_n, const TArg &arg_m, const TArg &arg_t,
25 const ImmShiftResults &shift_res) {
26 const auto is_32bit = (iflags & k32Bit) != 0u;
27 const bool is_wback = (iflags & kWBack) != 0u;
28 const bool is_index = (iflags & kIndex) != 0u;
29 const bool is_add = (iflags & kAdd) != 0u;
30
31 ExecFlagsSet eflags{0x0u};
32 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
33
34 if (!condition_passed) {
35 It::ITAdvance(ictx.pstates);
36 Pc::AdvanceInstr(ictx.pstates, is_32bit);
37 return Ok(ExecResult{eflags});
38 }
39
40 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
41 const auto rm = Reg::ReadRegister(ictx.pstates, arg_m.Get());
42
43 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(ictx.pstates);
44
45 auto offset = Alu32::Shift(rm, shift_res.type, shift_res.value,
46 (apsr & ApsrRegister::kCMsk) == ApsrRegister::kCMsk);
47
48 const u32 offset_address = is_add == true ? rn + offset : rn - offset;
49 const u32 address = is_index == true ? offset_address : rn;
50
51 TRY_ASSIGN(data, ExecResult, TOp::Read(ictx, address));
52
53 if (is_wback == true) {
54 Reg::WriteRegister(ictx.pstates, arg_n.Get(), offset_address);
55 }
56 if (arg_t.Get() == RegisterId::kPc) {
57 // When the given address was unaligend the behaviour is
58 // unpredtictable
59 if ((address & 0x3u) == 0u) {
60 It::ITAdvance(ictx.pstates);
61 TRY(ExecResult, (Pc::LoadWritePC(ictx.pstates, ictx.bus, data)));
62
63 return Ok(ExecResult{eflags});
64 } else {
65 return Err<ExecResult>(StatusCode::kScExecutorUnpredictable);
66 }
67 } else {
68 Reg::WriteRegister(ictx.pstates, arg_t.Get(), data);
69 }
70 It::ITAdvance(ictx.pstates);
71 Pc::AdvanceInstr(ictx.pstates, is_32bit);
72
73 return Ok(ExecResult{eflags});
74 }
