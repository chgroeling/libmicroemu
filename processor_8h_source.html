44 // Check if execution mode is thumb .. if not, raise usage fault and return true
45 static bool IsThumbModeOrRaise(TProcessorStates &pstates) {
46 const auto epsr = SReg::template ReadRegister<SpecialRegisterId::kEpsr>(pstates);
47 const bool is_thumb = (epsr & static_cast<u32>(EpsrRegister::kTMsk)) != 0U;
48
49 // if EPSR.T == 0, a UsageFault(‘Invalid State’) is taken
50 if (!is_thumb) {
51 auto cfsr = SReg::template ReadRegister<SpecialRegisterId::kCfsr>(pstates);
52 cfsr |= CfsrUsageFault::kInvStateMsk;
53 SReg::template WriteRegister<SpecialRegisterId::kCfsr>(pstates, cfsr);
54 ExcTrig::SetPending(pstates, ExceptionType::kUsageFault);
55 return false;
56 }
57 return true;
58 }

60 template <typename TDelegates>
61 static Result<StepFlagsSet> Step(TProcessorStates &pstates, TBus &bus, TDelegates &delegates) {
62 static constexpr u8 kRaw32BitMsk = static_cast<RawInstrFlagsSet>(RawInstrFlagsMsk::k32Bit);
63
64 StepFlagsSet step_flags{0U};
65
66 // *** FETCH ***
67 // The pc points to the current instruction +4 . Therefore decrement 4 to
68 // get the current address.
69
70 // Get the current program counter
71 me_adr_t pc_this_instr = static_cast<me_adr_t>(Reg::ReadPC(pstates) - 4U);
72
73 // Check for exceptions raised by peripherals before fetching the instruction
74 const auto exc_ctx_pre_fetch = ExceptionContext{pc_this_instr};
75 TRY_ASSIGN(if_pre_fetch_exception, StepFlagsSet,
76 Exc::template CheckExceptions<ExceptionPreFetch>(pstates, bus, exc_ctx_pre_fetch));
77
78 // If an asynchronous exception should be executed, the pc must be updated
79 // to the new instruction address.
80 if (if_pre_fetch_exception) {
81 pc_this_instr = static_cast<me_adr_t>(Reg::ReadPC(pstates) - 4U);
82 }
83
84 RawInstr raw_instr;
85
86 // Check if execution mode is thumb .. if not, raise usage fault and continue
87 if (IsThumbModeOrRaise(pstates)) {
88 auto r_raw_instr = TFetcher::Fetch(pstates, bus, pc_this_instr);
89 if (r_raw_instr.IsErr()) {
90 ErrorHandler(pstates, r_raw_instr, bus);
91 return Err<RawInstr, StepFlagsSet>(r_raw_instr);
92 }
93 raw_instr = r_raw_instr.content;
94 }
95
96 const auto exc_ctx_post_fetch = ExceptionContext{pc_this_instr};
97
98 // If an invalid state exception was raised, this will processed here
99 // Check for exceptions after fetching the instruction
100 TRY_ASSIGN(is_fetch_exception, StepFlagsSet,
101 Exc::template CheckExceptions<ExceptionPostFetch>(pstates, bus, exc_ctx_post_fetch));
102
103 if (is_fetch_exception) {
104 // special case: if the instruction fetch raises an exception, this cycle is
105 // considered as a NOP cycle.
106
107 // The pc was updated in the exception handler
108 step_flags |= static_cast<StepFlagsSet>(StepFlags::kStepOk);
109 return Ok<StepFlagsSet>(step_flags);
110 }
111
112 // *** DECODE ***
113 auto r_instr = TDecoder::Decode(pstates, raw_instr);
114 if (r_instr.IsErr()) {
115 ErrorHandler(pstates, r_instr, bus);
116 return Err<Instr, StepFlagsSet>(r_instr);
117 }
118 const auto &instr = r_instr.content;
119
120 // Decoder can not raise any exceptions
121#ifndef NDEBUG
122 static constexpr u8 k32BitMsk = static_cast<InstrFlagsSet>(InstrFlags::k32Bit);
123 // check if the raw instruction and the decoded instruction have the same width
124 assert(((((raw_instr.flags & kRaw32BitMsk) != 0U) && ((instr.nop.flags & k32BitMsk) != 0U)) ||
125 (((raw_instr.flags & kRaw32BitMsk) == 0U) && ((instr.nop.flags & k32BitMsk) == 0U))));
126#endif
127
128 // *** CALLBACK ***
129 if (delegates.IsPreExecSet()) {
130 const auto is_32bit = (raw_instr.flags & kRaw32BitMsk) == kRaw32BitMsk;
131 auto op_code = OpCode{raw_instr.low, raw_instr.high, is_32bit};
132 auto instr_to_mnemonic = InstrToMnemonic<TProcessorStates, It, Reg, SReg>(pstates, instr);
133 auto reg_access = RegAccessor<TProcessorStates, Reg, SReg>(pstates);
134 auto spec_reg_access = SpecialRegAccessor<TProcessorStates, SReg>(pstates);
135
136 auto emu_ctx =
137 EmuContext(pc_this_instr, op_code, instr_to_mnemonic, reg_access, spec_reg_access);
138 delegates.PreExec(emu_ctx);
139 }
140
141 // *** EXECUTE ***
142 const auto r_execute = TExecutor::Execute(pstates, bus, instr, delegates);
143
144 if (r_execute.IsErr()) {
145 ErrorHandler(pstates, r_execute, bus);
146 return Err<ExecResult, StepFlagsSet>(r_execute);
147 }
148
149 if (delegates.IsPostExecSet()) {
150 const auto is_32bit = (raw_instr.flags & kRaw32BitMsk) == kRaw32BitMsk;
151 auto op_code = OpCode{raw_instr.low, raw_instr.high, is_32bit};
152 auto instr_to_mnemonic = InstrToMnemonic<TProcessorStates, It, Reg, SReg>(pstates, instr);
153 auto reg_access = RegAccessor<TProcessorStates, Reg, SReg>(pstates);
154 auto spec_reg_access = SpecialRegAccessor<TProcessorStates, SReg>(pstates);
155 auto emu_ctx =
156 EmuContext(pc_this_instr, op_code, instr_to_mnemonic, reg_access, spec_reg_access);
157 delegates.PostExec(emu_ctx);
158 }
159
160 // *** Exit Conditions ***
161 const auto &execute_flags = r_execute.content.flags;
162
163 // all exit flags are mutually exclusive
164 if ((execute_flags & kExitFlagsMask) != 0U) {
165 if ((execute_flags & static_cast<ExecFlagsSet>(ExecFlags::kBkptReqExit)) != 0U) {
166 step_flags |= static_cast<StepFlagsSet>(StepFlags::kStepTerminationRequest);
167 return Ok<StepFlagsSet>(step_flags);
168 }
169 if ((execute_flags & static_cast<ExecFlagsSet>(ExecFlags::kSvcReqExit)) != 0U) {
170 step_flags |= static_cast<StepFlagsSet>(StepFlags::kStepTerminationRequest);
171 return Ok<StepFlagsSet>(step_flags);
172 }
173 if ((execute_flags & static_cast<ExecFlagsSet>(ExecFlags::kBkptReqErrorExit)) != 0U) {
174 return Err<StepFlagsSet>(StatusCode::kScExecutorExitWithError);
175 }
176 if ((execute_flags & static_cast<ExecFlagsSet>(ExecFlags::kSvcReqErrorExit)) != 0U) {
177 return Err<StepFlagsSet>(StatusCode::kScExecutorExitWithError);
178 }
179 }
180
181 // get the next instruction address
182 const auto exc_ctc_post_exec = ExceptionContext{pc_this_instr};
183
184 TRY(StepFlagsSet,
185 Exc::template CheckExceptions<ExceptionPostExecution>(pstates, bus, exc_ctc_post_exec));
186
187 step_flags |= static_cast<StepFlagsSet>(StepFlags::kStepOk);
188 return Ok<StepFlagsSet>(step_flags);
189 }

205 // TODO: move outside of the class
206 static Result<void> TakeReset(TProcessorStates &pstates, TBus &bus_) {
207 // see Armv7-M Architecture Reference Manual Issue E.e p.531
208 LOG_INFO(TLogger, "Resetting processor");
209
210 auto sys_ctrl = SReg::template ReadRegister<SpecialRegisterId::kSysCtrl>(pstates);
211
212 // CurrentMode = Mode_Thread;
213 sys_ctrl &= ~static_cast<u32>(SysCtrlRegister::kExecModeMsk);
214 sys_ctrl |= static_cast<u32>(SysCtrlRegister::kExecModeThread);
215
216 // PRIMASK<0> = '0'; /* priority mask cleared at reset */
217 // FAULTMASK<0> = '0'; /* fault mask cleared at reset */
218 // BASEPRI<7:0> = Zeros(8); /* base priority disabled at reset */
219
220 // if HaveFPExt() then /* initialize the Floating Point Extn */
221 if (false) {
222 // CONTROL<2:0> = '000'; /* FP inactive, stack is Main, thread is privileged */
223 // CPACR.cp10 = '00';
224 // CPACR.cp11 = '00';
225 // FPDSCR.AHP = '0';
226 // FPDSCR.DN = '0';
227 // FPDSCR.FZ = '0';
228 // FPDSCR.RMode = '00';
229 // FPCCR.ASPEN = '1';
230 // FPCCR.LSPEN = '1';
231 // FPCCR.LSPACT = '0';
232 // FPCAR = bits(32) UNKNOWN;
233 // FPFSR = bits(32) UNKNOWN;
234 // for i = 0 to 31
235 // S[i] = bits(32) UNKNOWN;
236 } else {
237 // CONTROL<1:0> = '00'; /* current stack is Main, thread is privileged */
238 sys_ctrl &= ~static_cast<u32>(SysCtrlRegister::kControlSpSelMsk);
239 sys_ctrl &= ~static_cast<u32>(SysCtrlRegister::kControlNPrivMsk);
240 }
241
242 SReg::template WriteRegister<SpecialRegisterId::kSysCtrl>(pstates, sys_ctrl);
243
244 // for i = 0 to 511 /* all exceptions Inactive */
245 // ExceptionActive[i] = '0';
246 Exc::InitDefaultExceptionStates(pstates);
247
248 // ResetSCSRegs(); /* catch-all function for System Control Space reset */
249 // ClearExclusiveLocal(ProcessorID()); /* Synchronization (LDREX* / STREX*) monitor support */
250 // ClearEventRegister(); /* see WFE instruction for more details */
251
252 // All registers are UNKNOWN
253 // for i = 0 to 12
254 // R[i] = bits(32) UNKNOWN;
255
256 // bits(32) vectortable = VTOR<31:7>:'0000000';
257 const auto vtor = SReg::template ReadRegister<SpecialRegisterId::kVtor>(pstates);
258 me_adr_t vectortable = vtor << 7 | 0x0U;
259
260 // SP_main = MemA_with_priv[vectortable, 4, AccType_VECTABLE] AND 0xFFFFFFFC<31:0>;
261 TRY_ASSIGN(sp_main, void,
262 bus_.template ReadOrRaise<u32>(pstates, vectortable,
263 BusExceptionType::kRaisePreciseDataBusError));
264 SReg::template WriteRegister<SpecialRegisterId::kSpMain>(pstates, sp_main);
265
266 // SP_process = ((bits(30) UNKNOWN):'00');
267 auto sp_process = SReg::template ReadRegister<SpecialRegisterId::kSpProcess>(pstates);
268 sp_process &= ~0x3U; // clear the two least significant bits
269 SReg::template WriteRegister<SpecialRegisterId::kSpProcess>(pstates, sp_process);
270
271 // LR = 0xFFFFFFFF<31:0>; /* preset to an illegal exception return value */
272 Reg::template WriteRegister<RegisterId::kLr>(pstates, 0xFFFFFFFFU);
273
274 // tmp = MemA_with_priv[vectortable+4, 4, AccType_VECTABLE];
275 TRY_ASSIGN(tmp, void,
276 bus_.template ReadOrRaise<u32>(pstates, vectortable + 0x4U,
277 BusExceptionType::kRaisePreciseDataBusError));
278
279 // tbit = tmp<0>;
280 auto tbit = tmp & 0x1U;
281
282 // APSR = bits(32) UNKNOWN; /* flags UNPREDICTABLE from reset */
283
284 // IPSR<8:0> = Zeros(9); /* Exception Number cleared */
285 auto ipsr = SReg::template ReadRegister<SpecialRegisterId::kIpsr>(pstates);
286 ipsr &= ~static_cast<u32>(IpsrRegister::kExceptionNumberMsk);
287 SReg::template WriteRegister<SpecialRegisterId::kIpsr>(pstates, ipsr);
288
289 auto epsr = SReg::template ReadRegister<SpecialRegisterId::kEpsr>(pstates);
290 // EPSR.T = tbit; /* T bit set from vector */
291 // EPSR.IT<7:0> = Zeros(8); /* IT/ICI bits cleared */
292 epsr &= ~static_cast<u32>(EpsrRegister::kItMsk) & // clear it bits
293 ~static_cast<u32>(EpsrRegister::kTMsk); // clear t bit
294 epsr |= tbit << EpsrRegister::kTPos;
295 SReg::template WriteRegister<SpecialRegisterId::kEpsr>(pstates, epsr);
296
297 // BranchTo(tmp AND 0xFFFFFFFE<31:0>); /* address of reset service routine */
298 auto entry_point = tmp & 0xFFFFFFFEU;
299 Pc::BranchTo(pstates, entry_point);
300 LOG_DEBUG(TLogger, "Set entry Point to 0x%08X / tbit:%i", entry_point, tbit);
301
302 // CSR.STKALIGN = '1'; <-- added as default
303 /* stack alignment is 8-byte aligned per default*/
304 auto ccr = SReg::template ReadRegister<SpecialRegisterId::kCcr>(pstates);
305 ccr |= static_cast<u32>(CcrRegister::kStkAlignMsk);
306 SReg::template WriteRegister<SpecialRegisterId::kCcr>(pstates, ccr);
307 LOG_TRACE(TLogger, "CSR: 0x%08X", ccr);
308
309#if IS_LOGLEVEL_TRACE_ENABLED == true // LOG TRACE apsr, ipsr, epsr, xpsr
310 {
311 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(pstates);
312 auto ipsr = SReg::template ReadRegister<SpecialRegisterId::kIpsr>(pstates);
313 auto epsr = SReg::template ReadRegister<SpecialRegisterId::kEpsr>(pstates);
314 auto xpsr = SReg::template ReadRegister<SpecialRegisterId::kXpsr>(pstates);
315 LOG_DEBUG(TLogger, "APSR: 0x%08X, IPSR: 0x%08X, EPSR: 0x%08X, XPSR: 0x%08X", apsr, ipsr, epsr,
316 xpsr);
317 }
318#endif
319 return Ok();
320 }
