46template <typename TOp, typename TInstrContext> class TernaryInstr {
47public:
48 using It = typename TInstrContext::It;
49 using Pc = typename TInstrContext::Pc;
50 using Reg = typename TInstrContext::Reg;
51 using SReg = typename TInstrContext::SReg;
52
53 template <typename TArg0, typename TArg1, typename TArg2, typename TArg3>
54 static Result<ExecResult> Call(TInstrContext &ictx, const InstrFlagsSet &iflags,
55 const TArg0 &arg_n, const TArg1 &arg_m, const TArg2 &arg_d,
56 const TArg3 &arg_a) {
57 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
58
59 ExecFlagsSet eflags{0x0U};
60 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
61
62 if (!condition_passed) {
63 It::ITAdvance(ictx.pstates);
64 Pc::AdvanceInstr(ictx.pstates, is_32bit);
65 return Ok(ExecResult{eflags});
66 }
67
68 const auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
69 const auto rm = Reg::ReadRegister(ictx.pstates, arg_m.Get());
70 const auto ra = Reg::ReadRegister(ictx.pstates, arg_a.Get());
71
72 auto result = TOp::Call(ictx, rn, rm, ra);
73
74 Reg::WriteRegister(ictx.pstates, arg_d.Get(), result.value);
75
76 if ((iflags & static_cast<InstrFlagsSet>(InstrFlags::kSetFlags)) != 0U) {
77 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(ictx.pstates);
78 // Clear N, Z, C, V flags
79 apsr &=
80 ~(ApsrRegister::kNMsk | ApsrRegister::kZMsk | ApsrRegister::kCMsk | ApsrRegister::kVMsk);
81
82 apsr |= ((result.value >> 31U) & 0x1U) << ApsrRegister::kNPos; // N
83 apsr |= Bm32::IsZeroBit(result.value) << ApsrRegister::kZPos; // Z
84 apsr |= (result.carry_out == true ? 1U : 0U) << ApsrRegister::kCPos; // C
85 apsr |= (result.overflow == true ? 1U : 0U) << ApsrRegister::kVPos; // V
86 SReg::template WriteRegister<SpecialRegisterId::kApsr>(ictx.pstates, apsr);
87 }
88 It::ITAdvance(ictx.pstates);
89 Pc::AdvanceInstr(ictx.pstates, is_32bit);
90
91 return Ok(ExecResult{eflags});
92 }
