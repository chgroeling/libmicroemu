27 static inline Result<void> BXWritePC(TProcessorStates &pstates, TBus &bus,
28 const me_adr_t &address) {
29 // see Armv7-M Architecture Reference Manual Issue E.e p.31
30
31 auto sys_ctrl = SReg::template ReadRegister<SpecialRegisterId::kSysCtrl>(pstates);
32
33 // TODO: Own functions for current mode and is_handler mode
34 const auto current_mode = sys_ctrl & SysCtrlRegister::kExecModeMsk;
35 const auto is_handler_mode = current_mode == SysCtrlRegister::kExecModeHandler;
36
37 // if CurrentMode == Mode_Handler && address<31:28> == ‘1111’ then
38 if (is_handler_mode && ((address & 0xF0000000U) == 0xF0000000U)) {
39 LOG_TRACE(TLogger, "BXWritePC (Exception Return): address=0x%08X, is_handler_mode=%d",
40 address, is_handler_mode);
41 // ExceptionReturn(address<27:0>);
42 return ExcRet::Return(pstates, bus, address & 0x0FFFFFFFU);
43 } else {
44 auto epsr = SReg::template ReadRegister<SpecialRegisterId::kEpsr>(pstates);
45
46 if ((address & 0x1) == 0U) {
47 LOG_ERROR(TLogger, "BXWritePC: Set wrong execution state");
48 }
49 // TODO: Add UsageFault exception
50
51 // EPSR.T = address<0>;
52 // if EPSR.T == 0, a UsageFault(‘Invalid State’) is taken on the next instruction
53 epsr &= ~EpsrRegister::kTMsk;
54 epsr |= (address & 0x1U) << EpsrRegister::kTPos;
55 SReg::template WriteRegister<SpecialRegisterId::kEpsr>(pstates, epsr);
56 BranchTo(pstates, address & (~0x1));
57 return Ok();
58 }
59 // should never reach here
60 }

62 static inline void BLXWritePC(TProcessorStates &pstates, const me_adr_t &address) {
63 // see Armv7-M Architecture Reference Manual Issue E.e p.31
64
65 if (address & 0x1) {
66 LOG_ERROR(TLogger, "BLXWritePC: Set wrong execution state");
67 }
68 // EPSR.T = address<0>;
69 // if EPSR.T == 0, a UsageFault(‘Invalid State’) is taken on the next instruction
70 auto epsr = SReg::template ReadRegister<SpecialRegisterId::kEpsr>(pstates);
71 epsr &= ~EpsrRegister::kTMsk;
72 epsr |= (address & 0x1U) << EpsrRegister::kTPos;
73 SReg::template WriteRegister<SpecialRegisterId::kEpsr>(pstates, epsr);
74 pstates.BranchTo(address & (~0x1U));
75 }
