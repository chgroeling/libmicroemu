171template <typename TOp, typename TInstrContext> class TernaryInstrWithShift {
172public:
173 using It = typename TInstrContext::It;
174 using Pc = typename TInstrContext::Pc;
175 using Reg = typename TInstrContext::Reg;
176 using SReg = typename TInstrContext::SReg;
177
178 template <typename TArg0, typename TArg1, typename TArg2>
179 static Result<ExecResult> Call(TInstrContext &ictx, const InstrFlagsSet &iflags,
180 const TArg0 &arg_n, const TArg1 &arg_m, const TArg2 &arg_d,
181 const ImmShiftResults &shift_res) {
182 const auto is_32bit = (iflags & static_cast<InstrFlagsSet>(InstrFlags::k32Bit)) != 0U;
183
184 ExecFlagsSet eflags{0x0U};
185 TRY_ASSIGN(condition_passed, ExecResult, It::ConditionPassed(ictx.pstates));
186
187 if (!condition_passed) {
188 It::ITAdvance(ictx.pstates);
189 Pc::AdvanceInstr(ictx.pstates, is_32bit);
190 return Ok(ExecResult{eflags});
191 }
192
193 auto rn = Reg::ReadRegister(ictx.pstates, arg_n.Get());
194 auto rm = Reg::ReadRegister(ictx.pstates, arg_m.Get());
195 auto result = TOp::Call(ictx, rn, rm, shift_res);
196
197 Reg::WriteRegister(ictx.pstates, arg_d.Get(), result.value);
198
199 if ((iflags & static_cast<InstrFlagsSet>(InstrFlags::kSetFlags)) != 0U) {
200 auto apsr = SReg::template ReadRegister<SpecialRegisterId::kApsr>(ictx.pstates);
201
202 // Clear N, Z, C, V flags
203 apsr &=
204 ~(ApsrRegister::kNMsk | ApsrRegister::kZMsk | ApsrRegister::kCMsk | ApsrRegister::kVMsk);
205
206 apsr |= ((result.value >> 31U) & 0x1U) << ApsrRegister::kNPos; // N
207 apsr |= Bm32::IsZeroBit(result.value) << ApsrRegister::kZPos; // Z
208 apsr |= (result.carry_out == true ? 1U : 0U) << ApsrRegister::kCPos; // C
209 apsr |= (result.overflow == true ? 1U : 0U) << ApsrRegister::kVPos; // V
210 SReg::template WriteRegister<SpecialRegisterId::kApsr>(ictx.pstates, apsr);
211 }
212 It::ITAdvance(ictx.pstates);
213 Pc::AdvanceInstr(ictx.pstates, is_32bit);
214
215 return Ok(ExecResult{eflags});
216 }
